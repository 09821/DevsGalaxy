local function createTab(name)
    local tabFrame = Instance.new("ScrollingFrame")

    tabFrame.Parent = MainContent
    
    local layout = Instance.new("UIListLayout")
    layout.Padding = UDim.new(0, 10)
    layout.Parent = tabFrame
    
    tabs[name] = tabFrame
    return tabFrame
end

local function switchTab(name)
    for tabName, tab in pairs(tabs) do
        tab.Visible = (tabName == name)
    end
    currentTab = name
end

-- Helper function to create checkbox buttons
local function createCheckbox(parent, text, position, callback)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 0, 35)
    frame.Position = position
    frame.BackgroundColor3 = Color3.fromRGB(60, 40, 90)
    frame.Parent = parent
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = frame
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, -40, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.TextSize = 13
    label.Font = Enum.Font.Gotham
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Position = UDim2.new(0, 10, 0, 0)
    label.Parent = frame
    
    local checkbox = Instance.new("TextButton")
    checkbox.Size = UDim2.new(0, 25, 0, 25)
    checkbox.Position = UDim2.new(1, -30, 0.5, -12.5)
    checkbox.BackgroundColor3 = Color3.fromRGB(40, 25, 60)
    checkbox.Text = ""
    checkbox.Parent = frame
    
    local checkCorner = Instance.new("UICorner")
    checkCorner.CornerRadius = UDim.new(0, 5)
    checkCorner.Parent = checkbox
    
    local checked = false
    
    checkbox.MouseButton1Click:Connect(function()
        checked = not checked
        checkbox.BackgroundColor3 = checked and Color3.fromRGB(100, 200, 100) or Color3.fromRGB(40, 25, 60)
        if callback then
            callback(checked)
        end
    end)
    
    return frame, checkbox
end

-- TP Brainrots V1
local TPV1Btn = createButton(MainTab, "TP Brainrots V1", UDim2.new(0, 0, 0, 135), function()
    if tweening then
        warn("Please wait for tween to finish!")
        return
    end
    
    if not selectedBrainrot then
        warn("Please select a brainrot first!")
        return
    end
    
    tweening = true
    
    local function isWaveNearby()
        local workspace = game:GetService("Workspace")
        local activeTsunamis = workspace:FindFirstChild("ActiveTsunamis")
        if not activeTsunamis then return false end
        
        for _, wave in pairs(activeTsunamis:GetChildren()) do
            if wave.Name:match("Wave") and wave:IsA("BasePart") then
                local distance = (wave.Position - HumanoidRootPart.Position).Magnitude
                if distance < 38 then
                    return true, wave
                end
            end
        end
        return false
    end
    
    local targetPos = selectedBrainrot:IsA("Model") and selectedBrainrot.PrimaryPart.Position or selectedBrainrot.Position
    
    spawn(function()
        while tweening and Character and HumanoidRootPart do
            local waveNear, wave = isWaveNearby()
            
            if waveNear then
                -- Go to lowest ground point
                HumanoidRootPart.CFrame = CFrame.new(HumanoidRootPart.Position.X, workspace.FallenPartsDestroyHeight + 50, HumanoidRootPart.Position.Z)
                
                -- Wait for wave to pass
                repeat
                    task.wait(0.1)
                    local dist = wave and (wave.Position - HumanoidRootPart.Position).Magnitude or 100
                until dist > 20
                
                -- Go up
                HumanoidRootPart.CFrame = HumanoidRootPart.CFrame + Vector3.new(0, 20, 0)
            else
                -- Move towards target
                local direction = (targetPos - HumanoidRootPart.Position).Unit
                HumanoidRootPart.CFrame = HumanoidRootPart.CFrame + direction * 2
                
                if (targetPos - HumanoidRootPart.Position).Magnitude < 5 then
                    tweening = false
                    break
                end
            end
            
            task.wait()
        end
    end)
end)

-- TP Brainrots V2
local TPV2Btn = createButton(MainTab, "TP Brainrots V2", UDim2.new(0, 0, 0, 180), function()
    if not selectedBrainrot then
        warn("Please select a brainrot first!")
        return
    end
    
    spawn(function()
        local workspace = game:GetService("Workspace")
        local activeTsunamis = workspace:FindFirstChild("ActiveTsunamis")
        
        -- Shrink wave hitboxes
        if activeTsunamis then
            for _, wave in pairs(activeTsunamis:GetChildren()) do
                if wave.Name:match("Wave") and wave:IsA("BasePart") then
                    wave.Size = Vector3.new(1, 1, 1)
                end
            end
        end
        
        -- Go underground
        HumanoidRootPart.CFrame = CFrame.new(HumanoidRootPart.Position.X, workspace.FallenPartsDestroyHeight + 27, HumanoidRootPart.Position.Z)
        task.wait(0.5)
        
        -- Float to brainrot
        local targetPos = selectedBrainrot:IsA("Model") and selectedBrainrot.PrimaryPart.Position or selectedBrainrot.Position
        
        while (targetPos - HumanoidRootPart.Position).Magnitude > 5 do
            local direction = (targetPos - HumanoidRootPart.Position).Unit
            HumanoidRootPart.CFrame = HumanoidRootPart.CFrame + direction * 2
            task.wait()
        end
        
        -- Activate proximity prompt
        local prompt = selectedBrainrot:FindFirstChildOfClass("ProximityPrompt", true)
        if prompt then
            fireproximityprompt(prompt)
        end
        
        task.wait(0.3)
        
        -- Go down
        HumanoidRootPart.CFrame = CFrame.new(HumanoidRootPart.Position.X, workspace.FallenPartsDestroyHeight + 27, HumanoidRootPart.Position.Z)
        task.wait(0.2)
        
        -- Move to the side
        HumanoidRootPart.CFrame = HumanoidRootPart.CFrame + Vector3.new(21, 0, 0)
        task.wait(0.2)
        
        -- Return to spawn
        local spawnLocation = workspace:FindFirstChild("SpawnLocation") or workspace:FindFirstChildOfClass("SpawnLocation", true)
        if spawnLocation then
            HumanoidRootPart.CFrame = spawnLocation.CFrame + Vector3.new(0, 5, 0)
        end
    end)

-- ESP System (Optimized to prevent lag)
local espUpdateInterval = 0.1
local lastEspUpdate = 0

RunService.RenderStepped:Connect(function()
    if not espEnabled or not selectedRarity then return end
    
    local currentTime = tick()
    if currentTime - lastEspUpdate < espUpdateInterval then return end
    lastEspUpdate = currentTime
    
    local workspace = game:GetService("Workspace")
    local activeBrainrots = workspace:FindFirstChild("ActiveBrainrots")
    if not activeBrainrots then return end
    
    local rarityFolder = activeBrainrots:FindFirstChild(selectedRarity)
    if not rarityFolder then return end
    
    -- Track existing brainrots
    local currentBrainrots = {}
    
    for _, brainrot in pairs(rarityFolder:GetChildren()) do
        local espId = brainrot:GetFullName()
        currentBrainrots[espId] = true
        
        if not espObjects[espId] then
            -- Create ESP
            local billboardGui = Instance.new("BillboardGui")
            billboardGui.Name = "ESP"
            billboardGui.AlwaysOnTop = true
            billboardGui.Size = UDim2.new(0, 200, 0, 50)
            billboardGui.StudsOffset = Vector3.new(0, 3, 0)
            billboardGui.Adornee = brainrot:IsA("Model") and brainrot.PrimaryPart or brainrot
            billboardGui.Parent = brainrot
            
            local textLabel = Instance.new("TextLabel")
            textLabel.Size = UDim2.new(1, 0, 1, 0)
            textLabel.BackgroundTransparency = 1
            textLabel.TextColor3 = Color3.fromRGB(255, 255, 0)
            textLabel.TextStrokeTransparency = 0.5
            textLabel.TextSize = 14
            textLabel.Font = Enum.Font.GothamBold
            textLabel.Parent = billboardGui
            
            espObjects[espId] = billboardGui
        end
        
        -- Update ESP text
        if espObjects[espId] and espObjects[espId].Parent then
            local pos = brainrot:IsA("Model") and brainrot.PrimaryPart and brainrot.PrimaryPart.Position or brainrot:IsA("BasePart") and brainrot.Position
            if pos then
                local distance = math.floor((HumanoidRootPart.Position - pos).Magnitude)
                local moneyValue = brainrot:FindFirstChild("BrainrotMoney") or brainrot:FindFirstChildOfClass("IntValue", true) or brainrot:FindFirstChildOfClass("NumberValue", true)
                local money = moneyValue and moneyValue.Value or "?"
                
                espObjects[espId].TextLabel.Text = string.format("%s\n%d studs | $%s\n[%s]", brainrot.Name, distance, tostring(money), selectedRarity)
            end
        end
    end
    
    -- Clean up removed brainrots
    for espId, espObj in pairs(espObjects) do
        if not currentBrainrots[espId] then
            if espObj then
                espObj:Destroy()
            end
            espObjects[espId] = nil
        end
    end
end)

-- Dragging System
local dragging
local dragInput
local dragStart
local startPos

local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

TitleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
